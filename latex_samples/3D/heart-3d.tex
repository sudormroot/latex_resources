\begin{tikzpicture}
  \begin{axis}[
      view={0}{10},
      axis equal,
      axis lines=none,
      colormap name =pink, 
    ]
    \addplot3[
      surf,
      shader=faceted,
      samples=50,
      domain=0:2*pi,y domain=0:2*pi,
      z buffer=sort,
      opacity=0.15
    ]
    (
      {(sin(deg(x)))^3*cos(deg(y))},
      {(sin(deg(x)))^3*sin(deg(y))},
      {(13*cos(deg(x))-5*cos(2*deg(x))-2*cos(3*deg(x))-cos(4*deg(x)))/16}
